def g(x):
    return x**4-4*x**2+5

def dg(x):
    return 4*x**3 - 8*x
In [4]:
def gradient_descent(initial_guess, multiplier, precision, max_iter):
    
    new_x = initial_guess
    
    x_list = [new_x]
    
    slope_list = [dg(new_x)]
    
    for n in range(max_iter):
        
        previous_x = new_x
        
        new_x = previous_x - multiplier*dg(previous_x)
        
        step_size = abs(new_x - previous_x)
        
        x_list.append(new_x)
        
        slope_list.append(dg(new_x))
        
        if step_size < precision:
            
            break
            
    return new_x,x_list,slope_list
